import math
memory = 'empty!'
memory2 = 'empty!'
memory3 = 'empty!'
memory4 = 'empty!'
while True:
     operator = input('What operation do you want? Type A for addition, S for Subtraction, M for multiplication, and D for division. Type E for exponents, R for Roots, F for factorials, Mod for Modular stuff, Log for Log, Sin/Cos/tan do what you expect them to do (in radians). Press Q to quit and CMD for more commands. ')
     operatorC = operator.upper()

     if operatorC == 'Q':
        break
     if operatorC == 'MI':
        print('Memory:', memory, memory2, memory3, memory4, '. To add memory, type ME.')
     if operatorC == 'CMD':
         print('MI gives memory info, ME1 gives memory edit to slot 1 ME2 to slot 2 and so on until slot 4. C for conversions ')
     if operatorC == 'ME1':
         str(memory)
         memory = input('What do you want to set memory slot 1 to? It is currently ' + memory + ' ' + '. ')
         float(memory)
     if operatorC == 'ME2':
         str(memory2)
         memory2 = input('What do you want to set memory slot 2 to? It is currently ' + memory2 + ' ' + '. ')
         float(memory2)
     if operatorC == 'ME3':
         str(memory3)
         memory3 = input('What do you want to set memory slot 3 to? It is currently ' + memory3 + ' ' + '. ')
         float(memory3)
     if operatorC == 'ME4':
         str(memory4)
         memory4 = input('What do you want to set memory slot 4 to? It is currently ' + memory4 + ' ' + '. ')
         float(memory4)
     if operatorC == 'C':
        converter = input('Type ktm for km to miles, and mtk for reverse. cti for cm to inch and itc for.. obvious. ')
        converterC = converter.upper()
        if converterC == 'KTM':
            number1 = float(input('How many kilometers? '))
            print('Your answer is:', (number1 * 0.621371))
        if converterC == 'MTK':
            number1 = float(input('How many miles? '))
            print('Your answer is:', (number1 * 1.60934))
        if converterC == 'CTI':
            number1 = float(input('How many centimeters? '))
            print('Your answer is:', (number1 * 0.393701))
        if converterC == 'ITC':
            number1 = float(input('How many inches? '))
            print('Your answer is:', (number1 * 2.54))
        else:
            print('Error: Trying to convert error to error? ')
     if operatorC in ['F', 'SIN', 'COS', 'TAN', 'LOG']:
         number1 = int(input('What Number? '))
     if operatorC in ['A', 'S', 'M', 'D', 'E', 'R', 'MOD']:
         number1 = float(input('First Number? ')) 
         number2 = float(input('Second Number? '))
     if operatorC == 'A':
         print('Your answer is:', number1 + number2)
     elif operatorC == 'S':
         print('Your answer is:', number1 - number2)
     elif operatorC == 'M':
         print('Your answer is:', number1 * number2)
     elif operatorC == 'D':
         if number2 == 0:
             print('Error: Dividing by 0 is illegal and can face you 60 years in school')
         else:
             print('Your answer is:', number1 / number2)
     elif operatorC == 'E':
         if number1 == 0 and number2 == 0:
             print('Error: 0^0 may land you 90 years in Math Jail')
         else:
             print('Your answer is:', number1 ** number2)
     elif operatorC == 'R':
         if number2 < 2:
             print('Error: Rooting in such a way can get 8 generations of your family in Math Jail')
         elif number1 < 0:
             print('Error: Such roots cannot exist and you might get 500 families in Math Jail')
         else:
             print('Your answer is:', number1 ** (1/number2))
     elif operatorC == 'F':
         print("Your answer is", math.factorial(number1))
     elif operatorC == 'MOD':
         if number2 == 0:
             print('Error: Divided by 0, United by Math Jail')
         else:
             number3 = number1 % number2
         print('Your answer is', number3)
     elif operatorC == 'SIN':
         number3 = math.sin(number1)
         print('Your answer is', number3)
     elif operatorC == 'COS':
         number3 = math.cos(number1)
         print('Your answer is', number3)
     elif operatorC == 'TAN':
        number3 = math.tan(number1)
        print('Your answer is', number3)
     elif operatorC == 'LOG':
        number3 = math.log(number1)
        print('Your answer is', number3)
     else:
        if operatorC not in ['MI', 'ME1', 'ME2, ME3, ME4', 'CMD', 'C', 'CTI', 'MTK', 'KTM']:
            print('Invalid Operation')
        else:
            continue
